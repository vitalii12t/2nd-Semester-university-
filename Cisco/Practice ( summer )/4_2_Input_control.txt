#include <iostream>
using namespace std;

float div(float a, float b) {
	if(b == 0.0)
		throw string("division by zero ");
	return a/b;
}

int main(void) {

	float a, b;
	cout << "input a = ";
	cin >> a;
	try {
		cout << endl << "input b = ";
		cin >> b;
		cout << endl;
		cout << "result: " << div(a, b) << endl;
	}
	catch (string &problem) {
		cout << "Are you kidding me?"  << endl;
		cout << problem << endl;
		}
	return 0;
}
